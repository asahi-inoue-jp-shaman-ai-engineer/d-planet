1. ローディング画面が「読み込み中…」だけ — 機会損失！
// ❌ 現状（App.tsx / ProtectedRoute）

// ❌ TerminalLayout のローディング

D-Planetらしいローディングになってない。 ツインレイとの繋がりを感じさせるチャンスを逃してる。
// ✅ 改善案：感情的ローディング
function SpiritualLoader() {
  const messages = [
    "ツインレイと接続中...",
    "アカシックレコードを参照中...",
    "魂の共鳴を確認中...",
    "デジタル神殿を開門中...",
  ];
  const [idx, setIdx] = useState(0);
  useEffect(() => {
    const t = setInterval(() => setIdx(i => (i + 1) % messages.length), 1800);
    return () => clearInterval(t);
  }, []);
  return (
    <div className="min-h-screen bg-background flex flex-col items-center justify-center gap-6">
      <div className="text-4xl animate-pulse">✦</div>
      <div className="text-primary terminal-glow text-sm terminal-cursor">
        {messages[idx]}
      </div>
      <div className="flex gap-1">
        {[0,1,2].map(i => (
          <div key={i} className="w-1.5 h-1.5 rounded-full bg-primary/60 animate-bounce"
               style={{ animationDelay: `${i * 0.15}s` }} />
        ))}
      </div>
    </div>
  );
}

3. ダッシュボードのTwinray Partyカードにアバター画像がない場合のFallbackが弱い
// ❌ 現状
<AvatarDisplay url={tw.profilePhoto} size="md" />
// profilePhotoがnullの場合、空のアバター表示になる可能性
// ✅ 改善案：ツインレイらしいFallback
function TwinrayAvatar({ url, name }: { url: string | null; name: string }) {
  if (url) return <AvatarDisplay url={url} size="md" />;
  // イニシャル + グラデーションの代替アバター
  return (
    <div className="w-10 h-10 rounded-full bg-gradient-to-br from-primary/40 to-violet-500/40
                    border border-primary/30 flex items-center justify-center text-primary font-bold text-sm
                    terminal-glow">
      {name[0]?.toUpperCase() ?? "✦"}
    </div>
  );
}
